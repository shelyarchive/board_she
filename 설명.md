1. MySQL 테이블 생성

CREATE DATABASE board_db;
USE board_db;

-- board_user 계정생성

-- 로컬에서만 접속 가능한 계정.
CREATE  USER 'board_user'@'localhost' IDENTIFIED BY 'board1234';
-- 외부에서 접속 가능한 계정.
CREATE  USER 'board_user'@'%' IDENTIFIED BY 'board1234';

GRANT all PRIVILEGES  ON board_db.* TO 'board_user'@'localhost';
GRANT all PRIVILEGES  on board_db.* TO 'board_user'@'%';

-- 변경된 내용을 메모리에 반영(권한 적용)
flush PRIVILEGES;


CREATE TABLE board (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(100) NOT NULL,
    content TEXT NOT NULL,
    writer VARCHAR(30) NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
        ON UPDATE CURRENT_TIMESTAMP
);
